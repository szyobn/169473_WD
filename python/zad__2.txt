def filtruj_zmiennoprzecinkowe(słownik):
    zmiennoprzecinkowe = []
    for klucz, wartość in słownik.items():
        if isinstance(klucz, float) and isinstance(wartość, float):
            zmiennoprzecinkowe.append((klucz, wartość))
    return zmiennoprzecinkowe
